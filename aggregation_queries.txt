SELECT od.calendaryear,
       od.monthnumberofyear,
       ds.salesterritorycountry,
       sum(f.orderquantity) total_quantity
FROM factinternetsales f
JOIN dimdate od ON od.datekey = f.orderdatekey
JOIN dimsalesterritory ds ON ds.salesterritorykey = f.salesterritorykey
GROUP BY od.monthnumberofyear,
         od.calendaryear,
         ds.salesterritorycountry
ORDER BY od.calendaryear,
         od.monthnumberofyear
